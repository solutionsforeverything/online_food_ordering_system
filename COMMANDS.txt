CREATE DATABASE Foodbite;
USE Foodbite;
CREATE TABLE users(name VARCHAR(25), email NVARCHAR(320) PRIMARY KEY, password VARCHAR(25));
CREATE TABLE admins(name VARCHAR(25), email NVARCHAR(320) PRIMARY KEY, password VARCHAR(25));
CREATE TABLE restaurants(name VARCHAR(25) PRIMARY KEY, earnings NUMERIC DEFAULT 0, rating NUMERIC DEFAULT 0, orders INT DEFAULT 0);
CREATE TABLE coupons(name VARCHAR(25) PRIMARY KEY, DEDUCTION NUMERIC);
CREATE TABLE cart(food VARCHAR(25), quantity DECIMAL, price DECIMAL);
CREATE TABLE rating(restaurant VARCHAR(25), rate DECIMAL);
